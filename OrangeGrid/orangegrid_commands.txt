# ============================================================================
# 7. CHECK GPU AVAILABILITY
# ============================================================================

# All GPU types on the cluster
condor_status -constraint 'TotalGpus > 0' -af CUDADeviceName CUDAGlobalMemoryMb | sort | uniq -c | sort -k3 -rn

# A100 80GB nodes (needed for 31B native)
condor_status -avail -constraint 'CUDAGlobalMemoryMb >= 81000' -af Machine TotalGpus CUDADeviceName CUDAGlobalMemoryMb

# L40S/A40 nodes (45GB — fits everything up to 20B native, 31B will auto-fallback)
condor_status -avail -constraint 'CUDAGlobalMemoryMb >= 45000' -af Machine TotalGpus CUDADeviceName CUDAGlobalMemoryMb

# GPU VRAM vs Model size (native fp16):
#   LLaMA 3.2 3B     ~7 GB    → any GPU
#   Gemma 4 E2B      ~5 GB    → any GPU
#   Gemma 4 E4B      ~9 GB    → any GPU
#   LLaMA 3.1 8B     ~17 GB   → RTX 6000+ (22GB)
#   Qwen 2.5 14B     ~29 GB   → L40S/A40+ (45GB)
#   GPT-OSS 20B      ~41 GB   → L40S/A40+ (45GB)
#   Gemma 4 26B-A4B  ~52 GB   → A100 80GB
#   Gemma 4 31B      ~62 GB   → A100 80GB
#
# If native doesn't fit, the engine auto-falls back: native → 8-bit → 4-bit

# ============================================================================
# 13. MODEL REFERENCE — Native-First Precision
# ============================================================================
#
# Key               ~VRAM (native)   Fits on           If OOM → fallback
# ─────────────────────────────────────────────────────────────────────────
# llama-3.2-3b       ~7 GB           Any GPU           n/a (always fits)
# gemma-4-e2b        ~5 GB           Any GPU           n/a
# gemma-4-e4b        ~9 GB           Any GPU           n/a
# llama-3.1-8b       ~17 GB          RTX 6000+         8bit → 4bit
# qwen-2.5-14b       ~29 GB          L40S/A40+         8bit → 4bit
# gpt-oss-20b        ~41 GB          L40S/A40+         8bit → 4bit
# gemma-4-26b        ~52 GB          A100 80GB         8bit → 4bit
# gemma-4-31b        ~62 GB          A100 80GB         8bit → 4bit
#
# The engine tries native fp16 first. If CUDA OOM, it automatically
# retries with 8-bit, then 4-bit quantization. The benchmark log
# reports which precision was actually used for each model.
